f = list(filter(str.isdigit, open("Task 7 input.txt", "r", encoding="utf-8").read().split()))
f = [int(item) for item in f]
print(f)
Schools = []
winthreeshold = 80
for x in range(len(f)):
    if x % 2 != 0:
        if f[x] >= winthreeshold:
            if f[x-1] in Schools:
                print()
            else:
                Schools.append((f[x-1]))

print(Schools)
print(list.sort(Schools))
