f = list(filter(str.isdigit, open("Task 8 input.txt", "r", encoding="utf-8").read().split()))
names = open("Task 8 input.txt", "r", encoding="utf-8").read().split()
rates = [int(item) for item in f[1::2]]
names = [''.join(x for x in i if x.isalpha()) for i in names]
names = list(filter(lambda x: x != "", names))
names = [x[0] + ' ' + x[1] for x in zip(names, names[1::])]
print(names)
for i, j in zip(names, rates):
    print(i, j)
