with open("input.txt", "r", encoding="utf-8") as f:
    a = list(map(lambda x: x.rstrip().split(), f.readlines()))
    a = sorted(a, key = lambda x: (-int(x[3]), int(x[2]), x[1]))
    y = 0
    for i in range(len(a)):
        if a[i][3]==a[0][3]:
            if y == int(a[i][2]):
                continue
            else:
                print(int(a[i][2]), end=' ')
                y = int(a[i][2])
